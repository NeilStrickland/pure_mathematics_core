 \fromSlide{10}{
  \item \ghost\vspace{-4ex}{\tiny\begin{align*}
   \sin(a)\cos(b)+\cos(a)\sin(b) &=
     \frac{e^{ia}-e^{-ia}}{2i}\frac{e^{ib}+e^{-ib}}{2} + 
     \frac{e^{ia}+e^{-ia}}{2}\frac{e^{ib}-e^{-ib}}{2i} \\
     &\fromSlide{11}{=  \frac{e^{i(a+b)}-e^{-i(a+b)}}{2i}}
       \fromSlide{12}{=\sin(a+b)}
  \end{align*}}
 }
