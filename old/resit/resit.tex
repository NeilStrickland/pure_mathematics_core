 \question % pfrac 3
  Convert the function $x^2/(x+2)^2$ to partial fraction form,
  and thus find $\displaystyle{\int\frac{x^2}{(x+2)^2}\,dx}$.
  \marks{6}

 \question % inverse 1
  Let $f\:(0,\infty)\xra{}(0,\infty)$ be given by
  $f(x)=\log(1+x^2)$.  Find a formula for $f^{-1}(x)$.  
  \marks{3}

 \question % composite 2
  If $f(x)=2x^3$, what is $(\log\circ f\circ\exp)(x)$?  Simplify your
  answer as much as possible.
  \marks{4}

 \question % log 1
  Find $\log_{1000}(\sqrt{10})$.
  \marks{2}

 \question % trig value
  Find $\tan(9999\pi/4)$,giving a brief justification for your answer.
  \marks{3}

 \question % identity 1
  Show that 
  $\displaystyle{\frac{1+\tanh(x)^2}{1-\tanh(x)^2} = \cosh(2x)}$.
  \marks{7}

 \question % diff rat 2
  Let $p$ and $q$ be nonzero constants, and put $y=(x^p-x^q)^{1/pq}$.
  Simplify $x(x^p-x^q)\frac{dy}{dx}$.
  \marks{6}

 \question % diff log 
  Find
  $\displaystyle{\frac{d}{dx}\log(x+2x^2+3x^3+4x^4)}$.
  \marks{2}

 \question % diff quotient
  Find 
  $\displaystyle{\frac{d}{dx}\left(\frac{x^2}{\log(x)}\right)}$.
  \marks{4}

 \question % diff packet
  Let $a$ be a constant.  Find $f'(x)$, where $f(x)=x^2\,e^{-1/(x+a)}$. 
  \marks{4}

 \question % int poly exp
  Find $\int (x^2-x+1) e^{-x}\, dx$
  \marks{6}

 \question % int exp trig 2
  Find $\int e^{-3x}\cos(4x)\,dx$
  \marks{6}

 \question % linear 1
  Find the general solution of the following system of equations:
  \begin{eqnarray*}
   w+x+y-z &=& -2 \\
   w+x-y-z &=& 0 \\
   w-x-y-z &=& 2.
  \end{eqnarray*}
  \marks{5}

 \question % inverse matrix
  Find the determinant and inverse of the matrix 
  \[ A=
     \left[\begin{array}{ccc}
      -3 & 3 & 4 \\ 4 & 0 & 3 \\ 3 & 3 & -4 
     \end{array}\right].
  \]
  (Hint: You will find it easiest to use the cofactor method.)
  \marks{8}

 \question % range 4
  Define $f\:[-2,1]\xra{}\R$ by $f(x)=x^2+2x+3$.  Find the range of $f$.
  \marks{4}

 \question % int trig
  Find $\int\sin(x)^2\cos(x)^2\,dx$.  (Hint: what are the formulae for
  $\sin(2\theta)$ and $\cos(2\theta)$?)
  \marks{8}

 \question % int subs 2
  By making a suitable substitution, find
  $\int\cos(x)\log(\sin(x))\,dx$.
  \marks{6}

 \question % int misc
  You may assume that
  $\displaystyle \int x^3 \log(x)^2\,dx = x^4(a\log(x)^2 + b\log(x) + c)$\\
  for some constants $a$, $b$ and $c$.  Find these constants, and thus
  evaluate $\int_1^e x^3\log(x)^2\, dx$.
  \marks{9}

 \question % determinant
  Find the determinant of the following matrix:
  \[ \left[\begin{array}{cccc}
      -a & a & 1 \\
      1 & 0 & -a \\
      a & a & -1
     \end{array}\right].
  \]
  For which values of $a$ is the matrix invertible?
  \marks{7}
