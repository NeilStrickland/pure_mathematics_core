 \question % pfrac 3
  Convert the function $(x^2+x+1)/(x+1)^2$ to partial fraction form,
  and thus find $\displaystyle{\int\frac{x^2+x+1}{(x+1)^2}\,dx}$.
  \marks{6}

 \question % inverse 1
  Let $f\:(0,\infty)\xra{}(1,\infty)$ be given by
  $f(x)=1/(1-e^{-x})$.  Find a formula for $f^{-1}(x)$.  
  \marks{4}

 \question % composite 2
  If $f(x)=2x+2$, what is $(\exp\circ f\circ\log)(x)$?  Simplify your
  answer as much as possible.
  \marks{4}

 \question % log 1
  Find $\log_{16}(1/2)$.
  \marks{2}

 \question % trig value
  Find $\sin(-7\pi/3)$.  (You should give an exact answer, not a
  decimal approximation.)
  \marks{2}

 \question % identity 1
  Show that 
  $\displaystyle{\frac{1+\tanh(x)^2}{1-\tanh(x)^2} = \cosh(2x)}$.
  \marks{6}

 \question % diff rat 2
  Let $a$, $n$ and $m$ be constants.  Find $f'(x)$, where
  $f(x)=(x^n+a)^m$. 
  \marks{2}

 \question % diff log 
  Find
  $\displaystyle{\frac{d}{dx}\log(1+x+x^2+x^3)}$.
  \marks{2}

 \question % diff quotient
  Find 
  $\displaystyle{\frac{d}{dx}\left(\frac{x}{\log(x)}\right)}$.
  \marks{3}

 \question % diff mobius
  Find 
  $\displaystyle{\frac{d}{dx}\left(\frac{3x+2}{4x+3}\right)}$.
  \marks{2}

 \question % diff packet
  Let $a$, $b$ and $\omega$ be constants.  Find $f'(x)$, where
  $f(x)=e^{-(x-a)^2/b}\sin(\omega x)$. 
  \marks{4}

 \question % int poly exp
  Find $\int x^2 e^x\, dx$
  \marks{5}

 \question % int exp trig 2
  Find $\int e^{3x}\sin(4x)\,dx$
  \marks{5}

 \question % linear 1
  Find the general solution of the following system of equations:
  \begin{eqnarray*}
   w+x+y+z &=& 0 \\
   w+x-y-z &=& 0 \\
   w-x+y-z &=& 0.
  \end{eqnarray*}
  \marks{7}

 \question % inverse matrix
  Find the inverse of the matrix 
  \[ \left[\begin{array}{ccc}
      1 & a & b \\ 0 & 1 & c \\ 0 & 0 & 1 
     \end{array}\right].
  \]
  \marks{6}

 \question % range 4
  Define $f\:(-1,1)\xra{}\R$ by $f(x)=x^2+2x+3$.  Find the range of $f$.
  \marks{4}

 \question % int trig
  Find $\int\sin(x)^2\cos(x)^2\,dx$
  \marks{7}

 \question % int subs 2
  By making a suitable substitution, find
  $\int\sin(x)\log(\cos(x))\,dx$.
  \marks{6}

 \question % int misc
  You may assume that
  $\displaystyle \int x^2 \log(x)^2\,dx = x^3(a\log(x)^2 + b\log(x) + c)$\\
  for some constants $a$, $b$ and $c$.  Find these constants, and thus
  evaluate $\int_1^e x^2\log(x)^2\, dx$.
  \marks{7}

 \question % determinant
  Find the determinant of the following matrix:
  \[ \left[\begin{array}{cccc}
      1 & a & 0 & 0 \\
      a & 1 & b & 0 \\
      0 & b & 1 & c \\
      0 & 0 & c & 1
     \end{array}\right].
  \]
  \marks{6}
